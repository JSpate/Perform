using Microsoft.Extensions.Logging;
using Perform;
using Perform.Model;
using System;
using System.Threading;
using System.Threading.Tasks;
using Math = Perform.Scripting.MathMethods;

public class GeneratedShowScript : IDisposable
{
    private readonly ShowScript _showScript;
    private readonly ILogger _logger;
    public const int MaxVal = 10;
    public int count = 0;
    public GeneratedShowScript(ShowScript showScript, ILogger logger)
    {
        _showScript = showScript;
        _logger = logger;
        _showScript.AddEventHandler(EventHandler_1, EventType.ButtonChange, 1, ButtonState.Down);
    }
    private Task EventHandler_1(CancellationToken cancellationToken) {
        for (var i = 0; i < MaxVal; ) {
        	if (cancellationToken.IsCancellationRequested)
        	{
        		          break;
                    }
            if (i == 5) {
                break;
            }
            if (i % 2 == 0) {
                continue;
            }
            _showScript.Set("DMX.Light1.r", i * 10);
            
        }
        try {
            _logger.LogInformation("Try") ;
        } catch (Exception ex) {
            _logger.LogError(ex) ;
        } finally {
            _logger.LogInformation("Finally") ;
        }
        return Task.CompletedTask;
    }
    public void Dispose() {
        _showScript.RemoveEventHandler(EventHandler_1, EventType.ButtonChange, 1, ButtonState.Down);
    }
}
